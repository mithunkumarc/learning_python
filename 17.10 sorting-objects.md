#### sorting objects in list

        >>> class Bob:
        ...     name = 'bob'
        ... 
        >>> class Ann:
        ...     name = 'ann'

        >>> class Zak:
                name = 'zak'

        >>> z = Zak()
        >>> b = Bob()
        >>> a = Ann()
        >>> L = [z,b,a]

        >>> L.sort(key = lambda x : x.name)

        >>> for n in L:
        ...     print(n.name)
        ... 

        output : 
                ann
                bob
                zak

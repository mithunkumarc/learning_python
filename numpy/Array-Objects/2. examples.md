#### matrix to ndarray

        import numpy as np
        m = np.matrix([[1,2], [3,4]])
        print(type(m))
        a = np.array(m)
        print(type(a))
        
        output :
        
        <class 'numpy.matrixlib.defmatrix.matrix'>
        <type 'numpy.ndarray'>


#### list to array and reshape 

        import numpy as np
        l1 = [1,2,3,4]
        l2 = [4,5,6,7]
        a = np.array([l1,l2])
        print(a)
        
        a1 = a.reshape(4,2)
        print('after reshape')
        print(a1)
        
        output : 
        [[1 2 3 4]
         [4 5 6 7]]
        after reshape
        [[1 2]
         [3 4]
         [4 5]
         [6 7]]
         
         
         
#### integer : datatype :
                        
        i  : i64                        //64bits
        i2 : i16                        //i2(bytes) : i16 (bits)
        i4 : i32
        i8 : i64
        
        import numpy as np
        int_a = np.array([1,2,3,4])
        print(int_a.dtype)
        int_a1 = np.array([1,2,3,4],dtype='i4')
        print(int_a1.dtype)
        
        output : 
        int64
        int32
        
#### float : datatype        

        f : f32
        f2 : f16
        f4 : f32
        f8 : f64
        
        import numpy as np
        f_a1 = np.array([1,2,3,4],dtype='f')            //dtype : optional
        f_a2 = np.array([1,2,3,4],dtype='f2')
        f_a3 = np.array([1,2,3,4],dtype='f4')
        f_a4 = np.array([1,2,3,4],dtype='f8')
        print(f_a1.dtype)
        print(f_a2.dtype)
        print(f_a3.dtype)
        print(f_a4.dtype)
        print(f_a1)
        
        output : 
        float32
        float16
        float32
        float64
        [1. 2. 3. 4.]
        
#### bool : datatype

#### string : datatype , size will be same as size of string with maximum letters       
        
        import numpy as np
        s_a1 = np.array(['o','t','t'])
        s_a2 = np.array(['on','tw','th'])
        s_a3 = np.array(['one','two','thr'])
        s_a4 = np.array(['one','two','thre'])
        s_a5 = np.array(['one','two','three'])
        print(s_a1.dtype)
        print(s_a2.dtype)
        print(s_a3.dtype)
        print(s_a4.dtype)
        print(s_a5.dtype)

        output : 
        |S1                     : s1 : number of letter 1
        |S2                            number of letter 2  
        |S3                            number of letter 3
        |S4                            number of letter 4
        |S5

        
        
#### default datatype for mixed data is string

        import numpy as np
        s_a1 = np.array(['one',True,123])
        print(s_a1.dtype)
        s_a2 = np.array([['one',True,123],['one',True,123]])
        print(s_a2.dtype)
        s_a3 = np.array([['one',True,123],['one',True,123],[123,'three',False]])
        print(s_a3.dtype)

        output : 
        |S4
        |S4
        |S5
